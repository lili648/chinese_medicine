## 六、代码仓库结构

```
chinese_medicine/
├── data/                         # 数据目录
│   ├── pubmed/                   # PubMed 英文文献
│   ├── chinese/                  # 中文医学文献
│   └── stopwords.txt             # 停用词表
├── dicts/                        # 实体词典
│   ├── disease_dict.txt
│   ├── drug_dict.txt
│   └── symptom_dict.txt
├── output/                       # 中间产物
├── frontend/                     # Vue 3 前端
│   └── src/
│       ├── App.vue               # 根组件（含导航）
│       ├── components/
│       │   ├── Search.vue        # 文献检索页
│       │   ├── Graph.vue         # 知识图谱可视化页
│       │   ├── Detail.vue        # 文献详情页
│       │   └── Admin.vue         # 管理后台页
│       └── api/client.js         # FastAPI 调用客户端
├── src/                          # Python 后端
│   ├── api/main.py               # FastAPI 入口（5 个 REST 路由）
│   ├── db/db_session.py          # SQLAlchemy 连接管理
│   ├── preprocessing/
│   │   ├── data_loader.py        # Article 数据类 + DataLoader
│   │   └── preprocessor.py       # Preprocessor + StopWords
│   ├── ner/
│   │   ├── entity_dict.py        # EntityDict 词典管理
│   │   └── entity_recognizer.py  # EntityRecognizer 识别引擎
│   └── knowledge_graph/
│       ├── db_schema.py          # DBSchema 表结构定义
│       ├── data_importer.py      # DataImporter 批量导入
│       ├── relation_builder.py   # RelationBuilder 关系构建
│       ├── graph_engine.py       # GraphEngine NetworkX 计算
│       └── query_api.py          # QueryService 查询封装
├── tests/                        # 测试目录
├── requirements.txt              # Python 依赖
├── README.md                     # 项目说明
└── .gitignore
```
